%define %ppointer_functions(TYPE,NAME)
%{
static TYPE *new_##NAME() { %}
%{ return new TYPE(); %}
%{}

%{ return new TYPE(value); %}
%{}

%{ if (*obj) delete *obj; %}
%{}

}

}
%}
